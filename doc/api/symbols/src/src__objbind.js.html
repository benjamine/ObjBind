354 	 * binds functions to events on target object (for multiple bindings, a map of functions can be used as argument)
355 	 * @param {string} eventType
356 	 * @param {Object} eventData
357 	 * @param {function()} handler
358 	 */
359     $o.OQHelper.prototype.bind = function(eventType, eventData, handler){
360         var evts = this.events || (this.events = {});
361         
362         var evMap;
363         if (typeof handler == 'undefined') {
364             handler = eventData;
365             eventData = null;
366         }
367         if (typeof eventType == 'object') {
368             evMap = eventType;
369         }
370         else {
371             evMap = {};
372             evMap[eventType] = handler;
373         }
374         
375         for (var evType in evMap) {
376             var evTypes = evType.split(' ');
377             for (var i = 0; i < evTypes.length; i++) {
378                 var evt = evts[evTypes[i]] || (evts[evTypes[i]] = []);
379                 
380                 if (this.obj instanceof Array &&
381                 (evTypes[i] === 'change' ||
382                 evTypes[i] === 'adding' ||
383                 evTypes[i] === 'added' ||
384                 evTypes[i] === 'removing' ||
385                 evTypes[i] === 'removed')) {
386                     this.interceptArrayModifiers();
387                 }
388                 else {
389                     if (evTypes[i] === 'change' ||
390                     evTypes[i] === 'propertyChanging' ||
391                     evTypes[i] === 'propertyChanged') {
392                         this.interceptProperties();
393                     }
394                 }
395                 if (typeof evMap[evType] != 'undefined') {
396                     var addToList = true;
397                     if (typeof evMap[evType] != 'function') {
398                         for (var ei = evt.length - 1; ei >= 0; ei--) {
399                             if (evt[ei] === evMap[evType]) {
400                                 addToList = false;
401                             }
402                         }
403                     }
404                     if (addToList) {
405                         evt.push({
406                             data: eventData,
407                             handler: evMap[evType]
408                         });
409                     }
410                 }
411             }
412         }
413         return this;
414     }

417 	 * Unbinds event handlers
418 	 * @param {Object} eventType
419 	 * @param {function()} [handler]
420 	 */
421     $o.OQHelper.prototype.unbind = function(eventType, handler){
422     
423         var evts = this.events;
424         if (!evts) {
425             return;
426         }
427         
428         if (typeof handler == 'undefined') {
429             if (typeof eventType === 'function') {
430                 handler = eventType;
431                 eventType = null;
432             }
433         }
434         
435         if (typeof eventType == 'string') {
436             var evTypes = eventType.split(' ');
437             for (var i = 0; i < evTypes.length; i++) {
438                 if (evTypes[i]) {
439                     var evt = evts[evTypes[i]];
440                     if (evt) {
441                         if (handler) {
442                             // remove handler for this event type
443                             evt.splice(0, evt.length);
444                             for (var j = 0, jl = evt.length; j < jl; j++) {
445                                 if (handler === evt[j].handler) {
446                                     evt.splice(j, 1);
447                                     j--;
448                                 }
449                             }
450                         }
451                         else {
452                             // remove all handlers for this event type
453                             evt.splice(0, evt.length);
454                         }
455                     }
456                 }
457             }
458         }
459         else {
460             if (handler) {
461                 // remove handler in all events
462                 for (var eName in evts) {
463                     var evt = evts[eName];
464                     // remove handler for this event type
465                     evt.splice(0, evt.length);
466                     for (var j = 0, jl = evt.length; j < jl; j++) {
467                         if (handler === evt[j].handler) {
468                             evt.splice(j, 1);
469                             j--;
470                         }
471                     }
472                 }
473             }
474             else {
475                 // remove all interceptors
476                 this.removeInterceptors();
477                 
478                 // remove all bindings
479                 delete this.events;
480                 
481             }
482         }
483         return this;
484     }

622     
623         var mode = $o.propertyInterceptMode();
624         if (mode === $o.propertyInterceptModes.UNKNOWN ||
625         mode === $o.propertyInterceptModes.NONE) {

627         }
628         
629         var current = {
630             getter: null,
631             setter: null
632         };
633         
634         if (mode === $o.propertyInterceptModes.DEFINESETTER) {
635             current.setter = this.obj.__lookupSetter__(name);
636             current.getter = this.obj.__lookupGetter__(name);
637         }
638         else {
639             try {
640                 var odesc = Object.getOwnPropertyDescriptor(this.obj, name);
641                 current.setter = odesc.get;
642                 current.getter = odesc.set;
643             } 
644             catch (err) {
645                 // no descriptor found
646             }
647         }
648         
649         if (!this.overrides) {
650             this.overrides = {};
651         }
652         var override = this.overrides[name];
653         
654         if (!override) {
655         
656             this.overrides[name] = {
657                 original: current
658             }
659             override = this.overrides[name];
660         }
661         
662         override.current = {
663             getter: getter,
664             setter: setter
665         };
666         setter.overriden = current.setter;
667         getter.overriden = current.getter;
668         
669         if (mode === $o.propertyInterceptModes.DEFINESETTER) {
670             if (setter) {
671                 try {
672                     this.obj.__defineSetter__(name, setter);
673                 } 
674                 catch (err) {
675                     if (console && console.log) {
676 						console.log('Object.__defineSetter__ for property \'' + name + '\' failed: ' + err);
677 					}
678                 }
679             }
680             if (getter) {
681                 try {
682                     this.obj.__defineGetter__(name, getter);
683                 } 
684                 catch (err) {
685                     if (console && console.log) {
686 						console.log('Object.__defineGetter__ for property \'' + name + '\' failed: ' + err);
687 					}
688                 }
689             }
690         }
691         else 
692             if (mode === $o.propertyInterceptModes.DEFINEPROPERTY ||
693             (mode === $o.propertyInterceptModes.DEFINEPROPERTYONLYDOM && $o.isDOMObject(this.obj))) {
694                 try {
695                     var desc = {};
696                     if (setter) {
697                         desc.set = setter;
698                     }
699                     if (getter) {
700                         desc.get = getter;
701                     }
702                     Object.defineProperty(this.obj, name, desc);
703                 } 
704                 catch (err) {
705 					if (console && console.log) {
706 						console.log('Object.defineProperty for property \'' + name + '\' failed: ' + err);
707 					}
708                 }
709             }
710             else {
711             // no property intercept supported
712             // use: $o(this.subject).prop('pname',value);  			
713             // 		$o(this.subject).prop({ pname1: value1, pname2: value2 });  			
714             //      $o(this.subject).prop('pname');  
715             }
716         
717         return this;
718     }
